# for 범위에 변수 넣는거 조심





# 2차 행렬 깊은 복사 주의

```java
// 1차행렬 깉은 복사
arr2 = arr0.clone();


// 얕은 복사
arr3 = arr1.clone();
arr3 = arr1;


// 깊은 복사 (for문 + clone 사용)
for(int i = 0; i < arr1.length; i++) {
	arr3[i] = arr1[i].clone();
}

// 깊은 복사 (2중 for문 사용)
for(int i = 0; i < arr1.length; i++) {
	for(int j = 0; j < 2; j++) {
		arr4[i][j] = arr1[i][j];
	}
}
```
